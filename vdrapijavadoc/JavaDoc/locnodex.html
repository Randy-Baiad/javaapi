Creating Standard Objects Using the LocateNode Command - Example

LocateNodeCommand locateNode = new LocateNodeCommand(); // create command.

// Identify the objects handling callbacks.
locateNode.setObjectWithNode (this); // Set caller to be object with node.
locateNode.setPromptForLogin(this);  // Object to use for prompting, which implements
                    ;                    // the IcanPromptForLogin interface.

// Create a list of key-value pairs representing the command parameters.
Vector keyValuePairs = new Vector();
keyValuePairs.addElement (new ICommand.KeyValuePair (CommandKeyType.DocumentServer, documentServer));

keyValuePairs.addElement (new ICommand.KeyValuePair (CommandKeyType.ReportId,reportID));
keyValuePairs.addElement (new ICommand.KeyValuePair (CommandKeyType.VersionId,versionID));
keyValuePairs.addElement (new ICommand.KeyValuePair (CommandKeyType.SectionId,sectionID));
keyValuePairs.addElement (new ICommand.KeyValuePair (CommandKeyType.Page,
Integer.toString(pageNumber)));

// Load the parameters into the LocateNode command and execute it.
try {
   locateNode.load(keyValuePairs);
   locateNode.execute();
}catch (ViewDirectException e) {
}
